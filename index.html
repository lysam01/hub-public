欢迎使用姨妈助手！

点击查看数据库初始化 SQL 
-- 在 Supabase SQL 编辑器中执行

create table if not exists public.daily_logs (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  date date not null,
  is_period boolean default false,
  period_start boolean default false,
  period_end boolean default false,
  flow text,
  temperature numeric(4,2),
  symptoms text[] default '{}',
  love boolean default false,
  air text,
  notes text,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique (user_id, date)
);

-- 行级安全策略
alter table public.daily_logs enable row level security;

drop policy if exists "用户只能查看自己的记录" on public.daily_logs;
create policy "用户只能查看自己的记录"
  on public.daily_logs
  for select
  using (auth.uid() = user_id);

drop policy if exists "用户只能插入自己的记录" on public.daily_logs;
create policy "用户只能插入自己的记录"
  on public.daily_logs
  for insert
  with check (auth.uid() = user_id);

drop policy if exists "用户只能更新自己的记录" on public.daily_logs;
create policy "用户只能更新自己的记录"
  on public.daily_logs
  for update
  using (auth.uid() = user_id);

drop policy if exists "用户只能删除自己的记录" on public.daily_logs;
create policy "用户只能删除自己的记录"
  on public.daily_logs
  for delete
  using (auth.uid() = user_id);

-- user_settings 表
create table if not exists public.user_settings (
  user_id uuid primary key references auth.users(id) on delete cascade,
  default_period_length integer default 7,
  default_cycle_length integer default 28,
  updated_at timestamptz default now()
);

alter table public.user_settings enable row level security;

drop policy if exists "用户只能查看自己的设置" on public.user_settings;
create policy "用户只能查看自己的设置"
  on public.user_settings
  for select
  using (auth.uid() = user_id);

drop policy if exists "用户只能更新自己的设置" on public.user_settings;
create policy "用户只能更新自己的设置"
  on public.user_settings
  for update
  using (auth.uid() = user_id);

drop policy if exists "用户只能插入自己的设置" on public.user_settings;
create policy "用户只能插入自己的设置"
  on public.user_settings
  for insert
  with check (auth.uid() = user_id);
